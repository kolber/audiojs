Usage

<!DOCTYPE html>
<head>
  <meta charset="utf-8">
  <title>audio.js</title>
  <script src="/js/audio.js"></script>
  <script>
    audiojs.events.ready(function() {
      audiojs.createAll({
        imageLocation: '/images/player-graphics.gif',
        swfLocation: '/swf/audiojs.swf'
      });
    });
  </script>
</head>
<body>
  <audio src="/mp3s/juicy.mp3" preload="auto"/></audio>
</body>
